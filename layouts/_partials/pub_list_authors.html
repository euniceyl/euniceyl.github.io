{{/* Publication listing authors — bold owner name; blue + asterisk when first or co-first author. */}} {{- $authors := .Params.authors -}} {{ if $authors }} {{- if not (reflect.IsSlice $authors) -}} {{- $authors = slice $authors -}} {{- end -}} {{ range $index, $raw := $authors }} {{- $slug := $raw | urlize -}} {{- $profile := partial "functions/get_author_profile" (dict "slug" $slug "title" $raw) -}} {{- $display := $profile.name_display | default $profile.title | default $raw -}} {{- $is_me := or (eq $slug "me") (eq $profile.slug "me") -}} {{- $author_note := "" -}} {{- if isset $.Params "author_notes" -}} {{- $author_note = index $.Params.author_notes $index | default "" | strings.TrimSpace -}} {{- end -}} {{- $is_me_highlighted := and $is_me (or (eq $index 0) (ne $author_note "")) -}} {{- if gt $index 0 }}, {{ end -}} {{- if $is_me_highlighted -}} {{ $display }}* {{- else if $is_me -}} {{ $display }} {{- else -}} {{ $display }} {{- end -}} {{- if and (not $is_me_highlighted) (isset $.Params "author_notes") -}} {{- with (index $.Params.author_notes $index) -}} * {{- end -}} {{- end -}} {{- end -}} {{- end -}}
